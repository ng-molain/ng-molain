.批量安装脚本
----
# use add
npm add ng-zorro-antd

# dependencies
npx npm install --save \
  lodash-es \
  @ngx-translate/core \

# devDependencies
npx npm install --save-dev \
  @angular-builders/custom-webpack \
  tailwindcss \
  polished \
  lodash @types/lodash-es \

----

"fontawesome-web": "^5.15.0",   # 这个是私有的
